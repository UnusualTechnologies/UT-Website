{{- /* Renders a spam-resistant email link. The address is base64-encoded into a data attribute so it never appears as plain text or a `mailto:` in the raw HTML — basic scrapers find nothing. js/main.js assembles it only on click. Params: email (required) — the address, e.g. "hello@unusualtechnologies.com" text — label shown before reveal (default "click to reveal email") class — extra CSS classes (e.g. "btn careers-apply-btn") mode — "text": first click reveals the address, second opens the mail client. "action": a single click opens the mail client without showing the address. "copy": copies the address to the clipboard and shows it below with an optional "open in your email app" link. Does not auto-open the mail client (avoids the OS handler dialog when none is configured). (default "text"). */ -}} {{- $email := .email -}} {{- if $email -}} {{- $text := .text | default "click to reveal email" -}} {{- $mode := .mode | default "text" -}} {{- $class := .class | default "" -}} {{- $enc := $email | base64Encode -}} {{ $text }} {{- end -}}
